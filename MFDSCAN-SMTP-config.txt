$SmtpServer = "smtp.office365.com"
$SmtpPort = 587
$EmailFrom = "youraddress@ahca.myflorida.com"
$EmailTo = "recipient@domain.com"
$Subject = "MFDSCAN ACCOUNT LOCKED OUT!"
$Body = "The MFDSCAN serice account has been locked out"

$Password = ConvertTo-SecureString "YourPasswordHere" -AsPlainText -Force
